469  public void MostrarInformacion()
470  {

472  else{
473  for(Jugador j: ListaJugadores)
474  {
475  if(j.toString().equals(comboJugadores.getSelectedItem().toString()))
476  {
477  lblDNIJugador.setText(j.getDNI());
478  lblNombreJugador.setText(j.getNombre()+" "+j.getApellido());
479  try {
480  lblEquipoJugador.setText(controller.ObtenerEquipoJugador(j));
481  } catch (RemoteException e) {
482  // TODO Auto-generated catch block
483  e.printStackTrace();
484  }
485  if(j.isCuota_pagada()) rdbtnPagada.setSelected(true); else rdbtnNoPagada.setSelected(true);
486  if(j.isLesionado()) rdbtnLesionado.setSelected(true); else rdbtnBien.setSelected(true);
487  if(j.isReconocimiento_medico()) rdbtnAprobado.setSelected(true);else rbtnSuspendido.setSelected(true);
488  }
489  }
490  }
